#include <iostream>
#include <string>
#include <vector>

struct MemberVariable {
    std::string type;
    std::string name;
    std::string access;
};

struct MemberFunction {
    std::string returnType;
    std::string name;
    std::string access;
    // Parameters are not handled in this simplified version
};

class ClassGenerator {
public:
    void runWizard();
    void generateClass();

private:
    std::string className;
    std::string baseClass;
    std::vector<MemberVariable> memberVariables;
    std::vector<MemberFunction> memberFunctions;
    bool generateDefaultConstructor;
    bool generateDestructor;

    void askBasicInfo();
    void askMemberVariables();
    void askMemberFunctions();
    void askAdditionalOptions();
};

void ClassGenerator::runWizard() {
    askBasicInfo();
    askMemberVariables();
    askMemberFunctions();
    askAdditionalOptions();
}

void ClassGenerator::askBasicInfo() {
    std::cout << "Enter class name: ";
    std::getline(std::cin, className);

    std::cout << "Enter base class (leave empty if none): ";
    std::getline(std::cin, baseClass);
}

void ClassGenerator::askMemberVariables() {
    int numVars;
    std::cout << "How many member variables? ";
    std::cin >> numVars;
    std::cin.ignore();

    for (int i = 0; i < numVars; ++i) {
        MemberVariable var;
        std::cout << "Enter member variable type: ";
        std::getline(std::cin, var.type);
        std::cout << "Enter member variable name: ";
        std::getline(std::cin, var.name);
        std::cout << "Enter access modifier (private/protected/public): ";
        std::getline(std::cin, var.access);
        memberVariables.push_back(var);
    }
}

void ClassGenerator::askMemberFunctions() {
    int numFuncs;
    std::cout << "How many member functions? ";
    std::cin >> numFuncs;
    std::cin.ignore();

    for (int i = 0; i < numFuncs; ++i) {
        MemberFunction func;
        std::cout << "Enter member function return type: ";
        std::getline(std::cin, func.returnType);
        std::cout << "Enter member function name: ";
        std::getline(std::cin, func.name);
        std::cout << "Enter access modifier (private/protected/public): ";
        std::getline(std::cin, func.access);
        memberFunctions.push_back(func);
    }
}

void ClassGenerator::askAdditionalOptions() {
    char response;
    std::cout << "Generate default constructor? (y/n): ";
    std::cin >> response;
    generateDefaultConstructor = (response == 'y' || response == 'Y');
    std::cin.ignore();

    std::cout << "Generate destructor? (y/n): ";
    std::cin >> response;
    generateDestructor = (response == 'y' || response == 'Y');
    std::cin.ignore();
}

void ClassGenerator::generateClass() {
    std::cout << "\nGenerated Class:\n\n";

    std::cout << "class " << className;
    if (!baseClass.empty()) {
        std::cout << " : public " << baseClass;
    }
    std::cout << " {\npublic:\n";
    if (generateDefaultConstructor) {
        std::cout << "    " << className << "() {}\n";
    }
    if (generateDestructor) {
        std::cout << "    ~" << className << "() {}\n";
    }

    for (const auto& func : memberFunctions) {
        if (func.access == "public") {
            std::cout << "    " << func.returnType << " " << func.name << "();\n";
        }
    }

    std::cout << "\nprivate:\n";
    for (const auto& var : memberVariables) {
        if (var.access == "private") {
            std::cout << "    " << var.type << " " << var.name << ";\n";
        }
    }

    for (const auto& func : memberFunctions) {
        if (func.access == "private") {
            std::cout << "    " << func.returnType << " " << func.name << "();\n";
        }
    }

    std::cout << "};\n";
}

int main() {
    ClassGenerator generator;
    generator.runWizard();
    generator.generateClass();
    return 0;
}
